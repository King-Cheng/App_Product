Python SDK调用示例

# 引入医疗AI SDK
from medical_ai_sdk import MedicalAISDK

# 初始化SDK
medical_ai = MedicalAISDK(
    api_key='your_api_key_here',
    secret='your_secret_here'
)

# 调用医学影像分析API
def analyze_medical_image(image_path):
    try:
        # 支持的影像类型: 'ct', 'mri', 'xray', 'ultrasound', 'pathology', 'fundus'
        result = medical_ai.image_analysis.analyze(
            image_path=image_path,
            image_type='ct',
            task_type='lung_nodule_detection'
        )
        
        print('分析结果:', result)
        return result
    except Exception as e:
        print('API调用失败:', str(e))
        raise

# 调用示例
result = analyze_medical_image('/path/to/ct_scan.jpg')
