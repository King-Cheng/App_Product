API接入指南

简单几步，将强大的科研分析能力集成到您的系统

Python SDK调用示例

# 引入医学AI SDK
from medical_ai import ResearchAssistant

# 初始化科研助手
assistant = ResearchAssistant(api_key="your_api_key_here")

# 智能文献分析示例
def analyze_literature(topic, time_range="last_5_years"):
    """
    分析特定主题的文献情况
    """
    try:
        # 执行文献分析
        results = assistant.literature_analysis(
            topic=topic,
            time_range=time_range,
            options={
                "include_trends": True,
                "include_keywords": True,
                "include_knowledge_graph": True,
                "max_papers": 5000
            }
        )
        
        # 打印分析结果摘要
        print(f"主题: {results['topic']}")
        print(f"相关文献数量: {results['total_papers']}")
        print(f"年度趋势: {results['yearly_trend']}")
        print(f"主要关键词: {results['top_keywords'][:10]}")
        
        return results
        
    except Exception as e:
        print(f"分析失败: {str(e)}")
        return None

# 研究假设生成示例
def generate_hypotheses(topic, related_topics=None):
    """
    基于研究主题生成研究假设
    """
    try:
        hypotheses = assistant.hypothesis_generation(
            topic=topic,
            related_topics=related_topics,
            options={
                "number_of_hypotheses": 5,
                "include_rationale": True
            }
        )
        
        return hypotheses
        
    except Exception as e:
        print(f"假设生成失败: {str(e)}")
        return None

# 使用示例
if __name__ == "__main__":
    # 分析特定主题的文献
    results = analyze_literature("人工智能在肺癌诊断中的应用")
    
    # 生成研究假设
    if results:
        # 基于分析结果中的关键词生成研究假设
        keywords = [kw['keyword'] for kw in results['top_keywords'][:5]]
        hypotheses = generate_hypotheses("人工智能在肺癌诊断中的应用", keywords)
        
        if hypotheses:
            print("\n生成的研究假设:")
            for i, hyp in enumerate(hypotheses['hypotheses'], 1):
                print(f"\n假设 {i}: {hyp['text']}")
                print(f"依据: {hyp['rationale']}")
