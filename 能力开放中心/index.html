简单高效的API接入体验

标准化的API接口，完善的文档支持，丰富的SDK工具，让您轻松集成AI能力

api_demo.js
// 引入医疗AI SDK
const MedicalAISDK = require('medical-ai-sdk');

// 初始化SDK
const medicalAI = new MedicalAISDK({
  apiKey: 'your_api_key_here',
  secret: 'your_secret_here'
});

// 调用医学文本分析API
async function analyzeMedicalText() {
  try {
    const result = await medicalAI.textAnalysis.analyze({
      text: '患者男，65岁，主诉：反复咳嗽、咳痰20年，加重伴气促1周。...',
      features: ['diagnosis', 'medication', 'risk_assessment']
    });
    
    console.log('分析结果:', result);
    return result;
  } catch (error) {
    console.error('API调用失败:', error.message);
    throw error;
  }
}

// 调用医学影像分析API
async function analyzeMedicalImage(imageUrl) {
  try {
    const result = await medicalAI.imageAnalysis.analyze({
      imageUrl: imageUrl,
      modality: 'chest_xray',
      tasks: ['detection', 'segmentation', 'report_generation']
    });
    
    console.log('影像分析结果:', result);
    return result;
  } catch (error) {
    console.error('影像分析失败:', error.message);
    throw error;
  }
}

// 调用示例
analyzeMedicalText();
// analyzeMedicalImage('https://example.com/chest_xray.jpg');
