API接入指南

简单几步，将医学教育AI能力集成到您的教学系统

JavaScript SDK调用示例

// 引入医学AI教育SDK
import { MedicalEducationSDK } from 'medical-ai-education';

// 初始化SDK
const educationSDK = new MedicalEducationSDK({
  apiKey: 'your_api_key_here',
  userId: 'student_123456'
});

// 获取个性化学习路径
export async function getPersonalizedLearningPath(studentInfo) {
  try {
    const path = await educationSDK.learningPath.getPersonalizedPath({
      educationLevel: studentInfo.level, // 教育阶段
      specialty: studentInfo.specialty,   // 专业方向
      learningGoals: studentInfo.goals,   // 学习目标
      knowledgeGaps: studentInfo.gaps     // 知识缺口
    });
    
    console.log('个性化学习路径:', path);
    return path;
  } catch (error) {
    console.error('获取学习路径失败:', error);
    return null;
  }
}

// 获取虚拟病例
export async function getVirtualCase(filters) {
  try {
    const virtualCase = await educationSDK.cases.getCase({
      specialty: filters.specialty,       // 专科
      difficulty: filters.difficulty,     // 难度级别
      caseType: filters.type,             // 病例类型
      keywords: filters.keywords          // 关键词
    });
    
    console.log('虚拟病例:', virtualCase);
    return virtualCase;
  } catch (error) {
    console.error('获取虚拟病例失败:', error);
    return null;
  }
}

// 提交病例答案并获取反馈
export async function submitCaseAnswer(caseId, answer) {
  try {
    const feedback = await educationSDK.cases.submitAnswer({
      caseId: caseId,
      diagnosis: answer.diagnosis,
      diagnosticBasis: answer.basis,
      treatmentPlan: answer.treatment
    });
    
    console.log('病例反馈:', feedback);
    return feedback;
  } catch (error) {
    console.error('提交答案失败:', error);
    return null;
  }
}

// 生成知识评测
export async function generateKnowledgeAssessment(params) {
  try {
    const assessment = await educationSDK.assessment.generate({
      topics: params.topics,              // 测试主题
      questionCount: params.count,        // 题目数量
      difficultyLevel: params.difficulty, // 难度级别
      questionTypes: params.types         // 题目类型
    });
    
    console.log('知识评测:', assessment);
    return assessment;
  } catch (error) {
    console.error('生成评测失败:', error);
    return null;
  }
}

// 示例用法
document.addEventListener('DOMContentLoaded', async () => {
  // 学生信息
  const studentInfo = {
    level: 'undergraduate',
    specialty: 'cardiology',
    goals: ['掌握心血管系统基础知识', '培养临床思维'],
    gaps: ['心电图解读', '心律失常诊断']
  };
  
  // 获取个性化学习路径
  const learningPath = await getPersonalizedLearningPath(studentInfo);
  
  // 如果获取到学习路径，显示给用户
  if (learningPath) {
    renderLearningPath(learningPath);
  }
});
